Joins : Here in this case we are going to combine rows of multiple tables on the basis of column. 

Inner join(join) : dono table ka common data 
left outer join :  dono table ka common data + left table ka pura data
right outer join :  dono table ka common data + right table ka pura data
full join :  dono table ka common data + dono table ka pura data
cross join : 
self join

------------------------

create table customer(cust_id int not null,cust_name varchar(45) not null);

insert into customer(cust_id,cust_name) values(101,"Andrew Anderson"),(102,"Peter Parker"),(103,"Simon Sim"),(104,"Jack Jackson");

select * from customer;

create table orders(order_id int not null,cust_id int not null,amount int not null);

insert into orders(order_id,cust_id,amount) values(1,101,5000),(3,103,7500);

select * from orders;

select customer.cust_id,customer.cust_name, orders.order_id,orders.amount from customer join orders on customer.cust_id = orders.cust_id;

select customer.cust_id,customer.cust_name, orders.order_id,orders.amount from customer inner join orders on customer.cust_id = orders.cust_id;

select customer.cust_id,customer.cust_name, orders.order_id,orders.amount from customer left outer join orders on customer.cust_id = orders.cust_id;

select customer.cust_id,customer.cust_name, orders.order_id,orders.amount from customer left join orders on customer.cust_id = orders.cust_id;

select customer.cust_id,customer.cust_name, orders.order_id,orders.amount from customer right join orders on customer.cust_id = orders.cust_id;

select customer.cust_id,customer.cust_name, orders.order_id,orders.amount from customer cross join orders;
